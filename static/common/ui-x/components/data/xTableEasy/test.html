xTableEasy 组件测试

1. 基本功能测试

<x-table-easy 
  :table-data="tableData" 
  :columns="columns" 
  row-key-field-name="id"
  border-around
></x-table-easy>

2. 自定义列宽和对齐方式测试

columns: [
  { key: 'id', title: 'ID', width: '80px', align: 'center' },
  { key: 'name', title: '姓名', width: '150px' },
  { key: 'age', title: '年龄', width: '80px', align: 'center' },
  { key: 'address', title: '地址' }
]

3. 自定义格式化测试

columns: [
  { key: 'id', title: 'ID' },
  { key: 'name', title: '姓名' },
  { 
    key: 'age', 
    title: '年龄',
    formatter: (row) => row.age + ' 岁'
  },
  { key: 'address', title: '地址' }
]

4. 空数据测试

<x-table-easy 
  :table-data="[]" 
  :columns="columns" 
  row-key-field-name="id"
  border-around
  empty-text="暂无数据，请添加"
></x-table-easy>

5. 最大高度测试

<x-table-easy 
  :table-data="largeData" 
  :columns="columns" 
  row-key-field-name="id"
  border-around
  max-height="300px"
></x-table-easy>
